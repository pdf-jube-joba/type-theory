\begin{itembox}[l]{項と型の定義}
  \begin{bnfgrammar}
    <type> ::= : T
    | <type-variable> : X
    | Arr <type> <type>
  \end{bnfgrammar}
  \begin{bnfgrammar}
    <term> ::= : t 
    | <term-variable> : x
    | Fun <term-variable> <type> <term>
    | App <term> <term>
  \end{bnfgrammar}
  \begin{bnfgrammar}
    <context> ::= : G
    | (<term-variable> , <type>) <context>
  \end{bnfgrammar}
\end{itembox}
